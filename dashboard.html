// ═══ DAILY SEARCH REMINDER ═══
function checkReminder(){
  const banner=document.getElementById("reminder-banner");
  const lastEl=document.getElementById("reminder-last");
  if(!banner||!lastEl)return;
  const ls=localStorage.getItem(LSK);
  const dismissed=sessionStorage.getItem("ff_reminder_dismissed");
  if(dismissed){banner.classList.remove("show");return}
  if(!ls){
    // Never searched
    lastEl.textContent="Nunca has buscado.";
    banner.classList.add("show");return;
  }
  const last=new Date(ls);
  const now=new Date();
  const diffH=(now-last)/(1000*60*60);
  if(diffH>=12){
    // More than 12 hours since last search
    const dStr=last.toLocaleDateString("es-CO",{day:"numeric",month:"short"});
    const tStr=last.toLocaleTimeString("es-CO",{hour:"2-digit",minute:"2-digit"});
    lastEl.textContent=`Última: ${dStr} ${tStr}`;
    banner.classList.add("show");
  }else{
    banner.classList.remove("show");
  }
}
function dismissReminder(){
  sessionStorage.setItem("ff_reminder_dismissed","1");
  document.getElementById("reminder-banner").classList.remove("show");
}
